Two Sum - Hash Map, Array

O(n2)


var twoSum = function(nums, target) {
    array=[]
    for (i=0; i< nums.length; i++){
        for(j=i+1; j<nums.length;j++){
            if(nums[i]+ nums[j]==target){
               array[0]= i;
               array[1]=j;
		return array;
            }
        }
    }
    
    
};
